3.4 示例项目
示例1：待办事项应用
swift
import CoreUI
import CoreFS

struct Todo: Identifiable, Codable {
    let id = UUID()
    var title: String
    var isCompleted = false
}

@main
struct TodoApp: App {
    var body: some Scene {
        WindowGroup {
            TodoListView()
        }
    }
}

struct TodoListView: View {
    @State private var todos: [Todo] = []
    @State private var newTodoTitle = ""
    
    var body: some View {
        NavigationView {
            List {
                ForEach($todos) { $todo in
                    TodoRow(todo: $todo)
                }
                .onDelete(perform: deleteTodo)
            }
            .navigationTitle("待办事项")
            .toolbar {
                ToolbarItem(placement: .navigationBarTrailing) {
                    Button(action: addTodo) {
                        Image(systemName: "plus")
                    }
                }
            }
        }
    }
    
    func addTodo() {
        let todo = Todo(title: newTodoTitle)
        todos.append(todo)
        newTodoTitle = ""
    }
    
    func deleteTodo(at offsets: IndexSet) {
        todos.remove(atOffsets: offsets)
    }
}

struct TodoRow: View {
    @Binding var todo: Todo
    
    var body: some View {
        HStack {
            Image(systemName: todo.isCompleted ? "checkmark.circle.fill" : "circle")
                .foregroundColor(todo.isCompleted ? .green : .gray)
                .onTapGesture {
                    todo.isCompleted.toggle()
                }
            Text(todo.title)
                .strikethrough(todo.isCompleted)
            Spacer()
        }
    }
}
示例2：AI图像生成应用
swift
import CoreUI
import CoreML

struct ImageGenView: View {
    @State private var prompt = ""
    @State private var generatedImage: Image?
    @State private var isGenerating = false
    
    var body: some View {
        VStack(spacing: 20) {
            TextField("输入描述...", text: $prompt)
                .textFieldStyle(RoundedBorderTextFieldStyle())
                .padding()
            
            Button(action: generateImage) {
                if isGenerating {
                    ProgressView()
                } else {
                    Text("生成图像")
                }
            }
            .disabled(prompt.isEmpty || isGenerating)
            
            if let image = generatedImage {
                image
                    .resizable()
                    .scaledToFit()
                    .cornerRadius(12)
                    .padding()
            }
            
            Spacer()
        }
        .padding()
    }
    
    func generateImage() {
        isGenerating = true
        Task {
            let model = try await CoreML.loadModel("CoreImageGen")
            let result = try await model.predict(prompt: prompt)
            generatedImage = Image(data: result.imageData)
            isGenerating = false
        }
    }
}
